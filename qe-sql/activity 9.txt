select c.customer_name, s.salesman_name from customer c join salesman s on c.salesman_id = s.ssalesman_id;
select c.customer_name, salesman.name, salesman.grade from customer inner join salesman on c.salesman_id = s.salesman_id where salesman.grade < 300 order by c.customer_name;
select c.customer_name, salesman_name, s.commission from customer inner join salesman on c.salesman_id = s.salesman_id where s.commission > 0.12;
select o.order_no, o.order_date, o.purchase_amount, c.customer_name, s.salesman_name, s.commission
from orders o
inner join customers c on o.customer_id = c.customer_id
inner join salesman s on c.salesman_id = s.salesman_id;
