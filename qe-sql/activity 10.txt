select o.*
from orders o
join salesman s on o.salesman_id = s.salesman_id
where s.customer_id = 3007;

select o.*
from orders o
join salesman s on o.salesman_id = s.salesman_id
where s.city = new york;

select count(*)
from customers
where grade > (select avg(grade) from customers where city = new york);

from orders o
join salesman s on o.salesman_id = s.salesman_id
where s.commission = (select max(commission) from salesman);
